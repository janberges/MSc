    %
    \draw [electron, double] (B) -- (I);

    \draw [fill=black!10!white] (T) -- (I) -- (O) -- (T);

    \foreach \point in {t, b, I, T, B, O} \fill (\point) circle (1.5pt);

    \node [below=3mm] at (T) {$\Gamma$};
