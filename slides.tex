            \only<2->{+ \alert<2>{U_{\vec k - \vec q}}}} \big]
            \\
            \chi_{\vec k}(\I \omega_n) &= \frac T {\alert<4> N}
            \sum_{\vec q m} \frac
                {\epsilon_{\vec q} - \mu + \chi_{\vec q}(\I \omega_m)}
